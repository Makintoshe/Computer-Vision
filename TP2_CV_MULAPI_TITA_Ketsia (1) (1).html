def numerotation(img):
    '''
    Balayage Vidéo
    '''
    # récupération des dimensions
    dimx, dimy = img.shape
    # numérotation
    k = 0 #ici je commence par initialiser la variable qui fiat le décompte !
    for i in range(dimx):
        for j in range(dimy):
            if img[i,j]==1 : 
                k += 1
                img[i,j] = k
    return img

def get_past(image, i, j): 
    '''
    Renvoie une liste contenant les coordonnées des voisins antérieur (passé)
    en 8-connexité !!!
    '''   
    if i == 0 and j == 0:  # pixel dans le coin supérieur gauche
        return []    
    gauche = [(i, j-1)] if image[i][j-1] else []  # voisin gauche    
    if i == 0:  # pixel sur la rangée supérieure
        return gauche    
    premiere_ligne = [(i-1, j_) for j_ in range(max(j-1, 0), min(j+2, image.shape[1])) if image[i-1, j_]]    
    if j == 0:  # pixel sur la première colonne
        return premiere_ligne
    return premiere_ligne + gauche

def chainage_contour(img):
    '''
    Cette fonction fait du chaînage de points de contour
    en traitement parallèle, afin de afin de ressortir la liste des
    contours d'une image.
    '''
    im_num = numerotation(img) # récupération de la numérotation de chaque pixel

    chain = [] # initialisation de la liste chaine
    
    ident = np.zeros(img.shape).astype("int") # récupération des dimensions de l'image
    xdim,ydim = ident.shape
    
    # Partie parcours vidéo de l'image :
    for i in range(xdim):
        for j in range(ydim):
            voisins = get_past(img,i,j) # voisins is NB : on récupère les voisins d'un pixel
            p = int(img[i,j]) # p est un pixel_contour
            if not p : # si il n'a pas de contours alors on passe au suivant
                continue
                
            if len(voisins)==0:
                chain.append([(i,j)]) # p : la coordonnée (i,j)
                ident[i,j]=len(chain)
            else:
                identifiant = ident[voisins[0]]
                chain[identifiant-1].append((i,j))
                ident[i,j]=identifiant 
                
                if(len(voisins)>1): # cas de fusions de contours
                    identifiants = [int(ident[v]) for v in voisins] # On récupère les contours(indices) , ex : [A,B,C]=[3,2,1]
                    identifiants.append(identifiant) # On a la liste des indices Chain = [[..],[..],[..]] id_x = [3,2]
                    id_min = min(identifiants) # on récupère l'id min de voisins : ex : id_min = 1
                    for id_i in identifiants:
                        if(id_i>id_min): # Si l'indice de id_min est plus grand que celle de l'indice principale
                            '''
                            Ci-dessous, LA LIGNE à MODIFIER !!!!!!
                            '''
                            #ident[i,j] = 1#ident[i,j] -= 1 # on réduit de 1 
                            for point_p in chain[id_i-1]: # 
                                ident[point_p] = id_min
                            chain[id_min-1].extend(chain[id_i-1])
                            chain.pop(id_i-1)
    return identifiant, chain, ident

e ) Difficultés rencontrées :¶

  • Concernant l'algorithme de la question (2), la difficulté était principalement au nivreau de la fusion, mais le reste n'est que logique. (Voir annexe)
